\textcolor{preprocessor}{#include <\hyperlink{a00386}{snes.h}>}

\textcolor{keyword}{extern} \textcolor{keywordtype}{char} m0, m0\_end, p0, p0\_end, t0, t0\_end;
\textcolor{keyword}{extern} \textcolor{keywordtype}{char} m1, m1\_end, p1, p1\_end, t1, t1\_end;
\textcolor{keyword}{extern} \textcolor{keywordtype}{char} m2, m2\_end, p2, p2\_end, t2, t2\_end;
\textcolor{keyword}{extern} \textcolor{keywordtype}{char} m3, m3\_end, p3, p3\_end, t3, t3\_end;

\textcolor{keywordtype}{unsigned} \textcolor{keywordtype}{char} sxbg1=0,sxbg2=0,sxbg3=0, flip=0;

\textcolor{comment}{//---------------------------------------------------------------------------------}
\textcolor{keywordtype}{int} main(\textcolor{keywordtype}{void}) \{
    \textcolor{comment}{// Initialize SNES }
    \hyperlink{a00350_a6047713bb5a73afd3cd1d77e336bcdad}{consoleInit}();
     
    \textcolor{comment}{// Copy tiles to VRAM, begin with adr=0x2000 for first BG }
    \textcolor{comment}{// ALso, always use specific mode0 palette entry (ppp)}
    \textcolor{comment}{// BG1 w/bits ppp=0 is entries 0-3 }
    \textcolor{comment}{// BG1 w/bits ppp=1 is entries 4-7 }
    \textcolor{comment}{// BG1 w/bits ppp=2 is entries 8-11 }
    \textcolor{comment}{// BG1 w/bits ppp=3 is entries 12-15 }
    \textcolor{comment}{// BG1 w/bits ppp=4 is entries 16-19 }
    \textcolor{comment}{// BG1 w/bits ppp=5 is entries 20-23 }
    \textcolor{comment}{// BG1 w/bits ppp=6 is entires 24-27 }
    \textcolor{comment}{// BG1 w/bits ppp=7 is entires 28-31 }
    \textcolor{comment}{// and then for BG2 ppp resets to 0 and resumes at 32... }
    \textcolor{comment}{// BG2 w/bits ppp=0 is entries 32-35 }
    \textcolor{comment}{// BG2 w/bits ppp=1 is entries 36-39 }
    \textcolor{comment}{// BG2 w/bits ppp=2 is entries 40-43 }
    \textcolor{comment}{// etc etc }
    \hyperlink{a00347_ae09a7ac09be78eef0f28a6c7aabc82ad}{bgInitTileSet}(0, &t0, &p0, 0, (&t0\_end - &t0), 8, BG\_4COLORS0, 0x2000);
    \hyperlink{a00347_ae09a7ac09be78eef0f28a6c7aabc82ad}{bgInitTileSet}(1, &t1, &p1, 0, (&t1\_end - &t1), 8, BG\_4COLORS0, 0x3000);
    \hyperlink{a00347_ae09a7ac09be78eef0f28a6c7aabc82ad}{bgInitTileSet}(2, &t2, &p2, 0, (&t2\_end - &t2), 8, BG\_4COLORS0, 0x4000);
    \hyperlink{a00347_ae09a7ac09be78eef0f28a6c7aabc82ad}{bgInitTileSet}(3, &t3, &p3, 0, (&t3\_end - &t3), 16, BG\_4COLORS0, 0x5000);

    \textcolor{comment}{// Copy Map to VRAM}
    \hyperlink{a00347_acc385296e159449a81d18f5e219c8032}{bgInitMapSet}(0, &m0, (&m0\_end - &m0),\hyperlink{a00347_a11724b6748ce0954e9b43819263537fc}{SC\_32x32}, 0x0000);
    \hyperlink{a00347_acc385296e159449a81d18f5e219c8032}{bgInitMapSet}(1, &m1, (&m1\_end - &m1),\hyperlink{a00347_a11724b6748ce0954e9b43819263537fc}{SC\_32x32}, 0x0400);
    \hyperlink{a00347_acc385296e159449a81d18f5e219c8032}{bgInitMapSet}(2, &m2, (&m2\_end - &m2),\hyperlink{a00347_a11724b6748ce0954e9b43819263537fc}{SC\_32x32}, 0x0800);
    \hyperlink{a00347_acc385296e159449a81d18f5e219c8032}{bgInitMapSet}(3, &m3, (&m3\_end - &m3),\hyperlink{a00347_a11724b6748ce0954e9b43819263537fc}{SC\_32x32}, 0x0C00);

    \textcolor{comment}{// Now Put in 16 color mode and disable other BGs except 1st one}
    \hyperlink{a00383_afd9e46ae627d055dd8c98a4b0ebb73b1}{setMode}(\hyperlink{a00347_a82bccfa80c8aa4ff5aa4e1416e0ade25}{BG\_MODE0},0);  
    
    \textcolor{comment}{// Wait for nothing :P}
    \textcolor{keywordflow}{while}(1) \{
        \textcolor{comment}{// Shcnage scrolling only each 3 frames}
        flip++;
        \textcolor{keywordflow}{if} (flip == 3) \{
            flip=0;
            sxbg3++;
            sxbg2+=2;
            sxbg1+=3;
            \hyperlink{a00347_aa5e0112ea2d4b514c2dddc6ed2c7c3c7}{bgSetScroll}(1,sxbg1,0);
            \hyperlink{a00347_aa5e0112ea2d4b514c2dddc6ed2c7c3c7}{bgSetScroll}(2,sxbg2,0);
            \hyperlink{a00347_aa5e0112ea2d4b514c2dddc6ed2c7c3c7}{bgSetScroll}(3,sxbg3,0);
        \}
        \hyperlink{a00356_a815e2626bd26d0e936d507a363ecd723}{WaitForVBlank}();
    \}
    \textcolor{keywordflow}{return} 0;
\}
